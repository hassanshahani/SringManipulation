DECLARE str1 : STRING
DECLARE str2 : STRING
DECLARE count : INTEGER
DECLARE s : INTEGER

str1 = ""
str2 = ""
count = 0

OUTPUT "Enter the STRING : "
INPUT = str1


str2 = str2 & MID(str1, Len(str1), 1)

For count = 1 To LENGHT(str1) - 1
  s = Len"str1" - count
  str2 = str2 & MID(str1, s, 1)
Next

OUTPUT "The inverse of your STRING is : " & str2
